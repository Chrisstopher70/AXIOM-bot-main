# Define main menu keyboard
main_menu_keyboard = [

# START command
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user = update.effective_user
    welcome_message = f"""
👔 *Welcome to AXIOM*, {user.first_name}!

I’m your memecoin trading assistant 🤖.
Choose an option below to get started:
"""
    await update.message.reply_text(welcome_message, parse_mode="Markdown", reply_markup=main_menu_markup)

# HANDLE menu buttons
async def handle_menu_selection(update: Update, context: ContextTypes.DEFAULT_TYPE):
    text = update.message.text

    if text == "💰 Connect Wallet":
        await update.message.reply_text("💰 ImpConnectort Wallet selected. Connecting Wallet...")
    elif text == "📤 Withdraw":
        await update.message.reply_text("📤 Withdraw selected. Coming soon!")
    elif text == "📊 PnL":
        await update.message.reply_text("📊 Your Profit & Loss data will show here.")
    elif text == "⚙️ Auto Trade":
        await update.message.reply_text("⚙️ Auto Trading enabled.")
    elif text == "📈 Trade":
        await update.message.reply_text("📈 Trade mode activated.")
    elif text == "🔥 Trending Coins":
        await handle_trending(update, context)
    else:
        await update.message.reply_text("❓ I didn’t understand that. Please use the menu.")

# TRENDING COINS handler
async def handle_trending(update: Update, context: ContextTypes.DEFAULT_TYPE):
    try:
        response = requests.get("https://api.coingecko.com/api/v3/search/trending")
        data = response.json()
        coins = data.get("coins", [])

        if not coins:
            await update.message.reply_text("No trending coins found at the moment 🔍")
            return

        message = "🔥 *Top Trending Coins on CoinGecko:*\n\n"
        for i, coin in enumerate(coins[:7], 1):
            item = coin['item']
            name = item.get('name', 'Unknown')
            symbol = item.get('symbol', 'N/A')
            market_rank = item.get('market_cap_rank', 'N/A')
            coin_id = item.get('id', '')
            address = item.get('contract_address', 'N/A')
            link = f"https://www.coingecko.com/en/coins/{coin_id}" if coin_id else "N/A"

            message += (
                f"*{i}. {name}* ({symbol})\n"
                f"• 🏅 Rank: {market_rank}\n"
                f"• 📬 Contract: `{address}`\n"
                f"• 🔗 [View on CoinGecko]({link})\n\n"
            )

        await update.message.reply_text(message, parse_mode="Markdown")

    except Exception as e:
        logger.error(f"Error fetching trending coins: {e}")
        await update.message.reply_text("⚠️ Failed to fetch trending coins.")

# MAIN
def main():
    app = ApplicationBuilder().token(TOKEN).build()

    app.add_handler(CommandHandler("start", start))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_menu_selection))

    print("✅ Bot is running...")
    app.run_polling()

if __name__ == "__main__":
    main()
